## 4. 组件Props详解

### 4.1 Alert 提示组件

```javascript
props: {
  type: { type: String, default: 'error' },    // 类型：error/warning/success
  tips: { type: String, default: '' },           // 提示内容
  showClose: { type: Boolean, default: true }  // 是否显示关闭按钮
}

emits: ['close']
```

### 4.2 bannerSwiper 轭播图组件
```javascript
props: {
  adBoxStyle: { type: String, default: '' },   // 广告容器样式
  showLocation: { type: String, default: '' }  // 展示位置标识
}
```

调用接口： `advertisementPage`
```javascript
api.advertisementPage({
  hotelCode: config.hotelCode,
  hotelGroupCode: config.hotelGroupCode
  showLocation: props.showLocation,
  clientTypes: channel,
  firstResult: 0,
  pageSize: 10
})
```

### 4.3 calendar 日历组件
```javascript
props: {
  checkInDate: { type: Array, default: [] },  // 入住日期数组
  checkOutDate: { type: Array, default: [] }, // 离店日期数组
  disabledDate: { type: Array, default: [] }, // 不可选日期数组
  startText: { type: String, default: '' },       // 开始日期文本
  endText: { type: String, default: '' },           // 结束日期文本
  price: { type: Number, default: 0 },              // 价格
  defaultPrice: { type: Number, default: 0 },       // 默认价格
  adults: { type: Number, default: 1 },        // 成人数量
  rooms: { type: Number, default: 1 },            // 房间数
  nights: { type: Number, default: 1 },            // 晚数
  orderInfo: { type: Object, default: {} },     // 订单信息
}

emits: ['change', 'check', 'confirm']
```

### 4.4 bottomDialog 底部对话框组件
```javascript
props: {
  show: { type: Boolean, default: false },  // 是否显示
  title: { type: String, default: '' },          // 标题
  content: { type: String, default: '' },        // 内容
  showClose: { type: Boolean, default: false } // 是否显示关闭
  confirmText: { type: String, default: '' }    // 确认按钮文本
  cancelText: { type: String, default: '' }    // 取消按钮文本
  closeText: { type: String, default: '' }     // 关闭时文本
}
emits: ['confirm', 'cancel', 'close']
```
### 4.5 bottomNav 底部导航组件(TabBar)
```javascript
props: {
  // 无props，使用固定配置
}
emits: ['tabChange']
```

### 4.6 ProductCard 产品卡片组件
```javascript
props: {
  data: { type: Object, default: {} },      // 产品数据
  showBottom: { type: Boolean, default: false }, // 是否显示底部
  showAdTag: { type: Boolean, default: true },  // 是否显示广告标签
  adTag: { type: String, default: '' }        // 广告标签
}
emits: ['click', 'favorite', 'share']
```

### 4.7 ProductCardList 产品卡片列表组件
```javascript
props: {
  list: { type: Array, default: [] },       // 产品列表

  loading: { type: Boolean, default: false }, // 加载状态
```

### 4.8 search/AutoComplete 自动完成组件
```javascript
props: {
  placeholder: { type: String, default: '请输入关键词' },
```

### 4.9 Search/SearchBar 搜索栏组件
```javascript
props: {
```
